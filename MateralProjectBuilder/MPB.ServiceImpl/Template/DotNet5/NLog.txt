<?xml version="1.0" encoding="utf-8" ?>
<nlog xmlns="http://www.nlog-project.org/schemas/NLog.xsd"
      xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
      autoReload="true"
      internalLogLevel="Info"
      internalLogFile="Logs/${var:AppName}/InternalNLog.log">
  <extensions>
    <add assembly="NLog.Web.AspNetCore"/>
  </extensions>
  <targets>
    <target xsi:type="File" name="AllFile" maxArchiveDays="${var:MaxLogFileSaveDays}" fileName="Logs/${var:AppName}/AllFile-${shortdate}.log"
            layout="${longdate}|${uppercase:${level}}|${logger}|${message} ${exception:format=tostring}" />
    <target xsi:type="File" name="ErrorFile" maxArchiveDays="${var:MaxLogFileSaveDays}" fileName="Logs/${var:AppName}/ErrorFile-${shortdate}.log"
            layout="${longdate}|${uppercase:${level}}|${logger}|${message} ${exception:format=tostring}" />
    <target xsi:type="File" name="DebugFile" maxArchiveDays="${var:MaxLogFileSaveDays}" fileName="Logs/${var:AppName}/DebugFile-${shortdate}.log"
            layout="${longdate}|${uppercase:${level}}|${logger}|${message} ${exception:format=tostring}" />
    <target xsi:type="File" name="HostingFile" maxArchiveDays="${var:MaxLogFileSaveDays}" fileName="Logs/${var:AppName}/HostingFile-${shortdate}.log"
            layout="${longdate}|${uppercase:${level}}|${logger}|${message} ${exception:format=tostring}" />
    <target xsi:type="Console" name="LifetimeConsole"
            layout="${longdate}|${uppercase:${level}}|${logger}|${message} ${exception:format=tostring}" />
  </targets>
  <rules>
    <!--Output hosting lifetime messages to console target for faster startup detection -->
    <logger name="Microsoft.Hosting.Lifetime" minlevel="Info" writeTo="LifetimeConsole, HostingFile" final="true" />
    <!--Skip non-critical Microsoft logs and so log only own logs (BlackHole) -->
    <logger name="Microsoft.*" maxlevel="Info" final="true" />
    <logger name="System.Net.Http.*" maxlevel="Info" final="true" />
    <!--<logger name="*" minlevel="Trace" writeTo="AllFile" />-->
    <logger name="*" minlevel="Info"  writeTo="LifetimeConsole" />
    <logger name="*" minlevel="Debug" maxlevel="Debug" writeTo="DebugFile, LifetimeConsole" />
    <logger name="*" minlevel="Error" writeTo="ErrorFile, LifetimeConsole" />
  </rules>
</nlog>
