using Materal.EnumHelper;
using Materal.TTA.SqlServerRepository.Model;
using Microsoft.Extensions.Logging;

namespace ${ProjectName}.Common.ConfigModels
{
    public class NLogConfigModel : BaseConfigModel
    {
        protected override string GetConfigKey() => "NLogSetting";
        /// <summary>
        /// 最小日志等级
        /// </summary>
        public LogLevel MinLogLevel
        {
            get
            {
                string value = GetConfigValue(nameof(MinLogLevel));
                LogLevel result = EnumManager.Parse<LogLevel>(value);
                return result;
            }
        }
        /// <summary>
        /// 最大文件保存天数
        /// </summary>
        public string MaxLogFileSaveDays => GetConfigValue(nameof(MaxLogFileSaveDays));
    }
}
