using AutoMapper;
using ${ProjectName}.PresentationModel.${RelativeNamespace}.${Name};
using ${ProjectName}.Service.${RelativeNamespace}.Models.${Name};

namespace ${ProjectName}.WebAPI.AutoMapperProfile.${RelativeNamespace}
{
    /// <summary>
    /// AutoMapper配置
    /// </summary>
    public sealed class ${Name}Profile : Profile
    {
        /// <summary>
        /// AutoMapper配置
        /// </summary>
        public ${Name}Profile()
        {
            CreateMap<Add${Name}RequestModel, Add${Name}Model>();
            CreateMap<Edit${Name}RequestModel, Edit${Name}Model>();
            CreateMap<Query${Name}FilterRequestModel, Query${Name}FilterModel>();
        }
    }
}
