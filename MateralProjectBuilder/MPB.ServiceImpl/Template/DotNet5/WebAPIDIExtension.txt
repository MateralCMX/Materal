using ${ProjectName}.DependencyInjection;
using Microsoft.Extensions.DependencyInjection;
using System.Reflection;

namespace ${ProjectName}.WebAPI
{
    /// <summary>
    /// WebAPI依赖注入扩展
    /// </summary>
    public static class ${ProjectName}WebAPIDIExtension
    {
        /// <summary>
        /// 添加WebAPI服务
        /// </summary>
        /// <param name="services"></param>
        public static void Add${ProjectName}WebAPIServices(this IServiceCollection services)
        {
            services.Add${ProjectName}Services();
            services.AddAutoMapperService(
                Assembly.Load("${ProjectName}.ServiceImpl"),
                Assembly.Load("${ProjectName}.WebAPI"));
        }
    }
}
